def api_runner():
    global df  # Allow modification of the global DataFrame
    url = 'https://pro-api.coinmarketcap.com/v1/cryptocurrency/listings/latest'
    parameters = {
        'start': '1',
        'limit': '15',
        'convert': 'USD'
    }
    headers = {
        'Accepts': 'application/json',
        'X-CMC_PRO_API_KEY': '7b5a52ca-4c21-446b-9b37-9a4bd285dd7a',  # Replace with your actual API key
    }

    session = Session()
    session.headers.update(headers)

    try:
        response = session.get(url, params=parameters)
        data = json.loads(response.text)

        # Check if the response contains 'data'
        if 'data' in data:
            # Normalize JSON data to a DataFrame
            df2 = pd.json_normalize(data['data'])
            df2['timestamp'] = pd.to_datetime('now')

            # Append to the global DataFrame
            df = pd.concat([df, df2], ignore_index=True)
            print("Data fetched and appended successfully!")
        else:
            print("No 'data' field found in response:", data)

    except (ConnectionError, Timeout, TooManyRedirects) as e:
        print("Error:", e)
